<Tab.Navigator
  screenOptions={{

          name={focused ? 'home' : 'home-outline'} 
          size={28}                    // ✅ Büyük icon

        />
      ),
    }}
  />

        <Ionicons

          size={28}                    // ✅ Büyük icon
